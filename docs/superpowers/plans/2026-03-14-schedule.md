# Schedule Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement scheduled backup functionality using node-cron for automated backups with retention policy

**Architecture:**
- **Schedule Module**: Manage scheduled backup tasks with cron expressions, enable/disable tasks, configure retention days
- **Schedule Store**: Persist schedule configuration to JSON file
- **CLI Commands**: Enable/disable schedules, set retention, list all scheduled tasks
- **Retention Policy**: Automatically delete backups older than specified retention days

**Tech Stack:**
- node-cron for cron-based scheduling
- fs/promises for persistence
- fs-extra for directory operations
- Existing backup and metadata modules
- TypeScript with strict mode

// packages/core/src/schedule/store.ts
import fs from 'fs/promises';
import path from 'path';
import type { ScheduleTask } from '../types/scheduler.js';

export interface ScheduleStore {
  add(schedule: ScheduleTask): Promise<void>;
  get(id: string): Promise<ScheduleTask | null>;
  list(): Promise<ScheduleTask[]>;
  update(id: string, updates: Partial<ScheduleTask>): Promise<void>;
  delete(id: string): Promise<void>;
  clear(): Promise<void>;
}

const SCHEDULE_FILE = 'schedules.json';

export function getDefaultScheduleIndexPath(backupDir: string): string {
  return path.join(backupDir, SCHEDULE_FILE);
}

export function createScheduleStore(indexPath: string): ScheduleStore {
  let schedules: Record<string, ScheduleTask> = {};
  let loaded = false;

  async function load(): Promise<void> {
    if (loaded) return;

    try {
      const content = await fs.readFile(indexPath, 'utf-8');
      const data = JSON.parse(content);
      schedules = data.schedules || {};
    } catch {
      schedules = {};
    }
    loaded = true;
  }

  async function save(): Promise<void> {
    const content = JSON.stringify({ schedules }, null, 2);
    await fs.mkdir(path.dirname(indexPath), { recursive: true });
    await fs.writeFile(indexPath, content, 'utf-8');
  }

  return {
    async add(schedule: ScheduleTask): Promise<void> {
      await load();
      schedules[schedule.id] = schedule;
      await save();
    },

    async get(id: string): Promise<ScheduleTask | null> {
      await load();
      return schedules[id] || null;
    },

    async list(): Promise<ScheduleTask[]> {
      await load();
      return Object.values(schedules);
    },

    async update(id: string, updates: Partial<ScheduleTask>): Promise<void> {
      await load();
      if (!schedules[id]) {
        throw new Error(`Schedule not found: ${id}`);
      }
      schedules[id] = { ...schedules[id], ...updates };
      await save();
    },

    async delete(id: string): Promise<void> {
      await load();
      delete schedules[id];
      await save();
    },

    async clear(): Promise<void> {
      schedules = {};
      await save();
    },
  };
}
```

// packages/core/src/schedule/schedule.ts
import { CronJob, CronTime } from 'cron';
import { createScheduleStore, getDefaultScheduleIndexPath, type ScheduleStore } from './store.js';
import { createMetadataStore, getDefaultMetadataIndexPath } from '../metadata-store/index.js';
import { backup } from '../backup/index.js';
import type { BackupLevel, BackupMetadata } from '../types/backup.js';
import type { ScheduleTask } from '../types/scheduler.js';
import crypto from 'crypto';

export interface ScheduleOptions {
  backupDir?: string;
}

export interface EnableScheduleOptions {
  name: string;
  level: BackupLevel;
  cron: string;
  retainDays: number;
  backupDir?: string;
}

// Active cron jobs
const activeJobs = new Map<string, CronJob>();

export async function enableSchedule(
  options: EnableScheduleOptions,
): Promise<ScheduleTask> {
  const backupDir = options.backupDir || '.clawguard/backups';
  const scheduleStore = createScheduleStore(getDefaultScheduleIndexPath(backupDir));

  // Generate schedule ID
  const id = generateScheduleId();

  // Calculate next run time based on cron expression
  const cronTime = new CronTime(options.cron);
  const nextRun = Date.now() + (cronTime.source?.interval || 86400000);

  const schedule: ScheduleTask = {
    id,
    name: options.name,
    cron: options.cron,
    lastRun: null,
    nextRun,
    enabled: true,
    level: options.level,
    retainDays: options.retainDays,
  };

  // Store schedule
  await scheduleStore.add(schedule);

  // Start cron job
  startCronJob(id, schedule, backupDir);

  return schedule;
}

export async function disableSchedule(
  id: string,
  options?: ScheduleOptions,
): Promise<ScheduleTask> {
  const backupDir = options?.backupDir || '.clawguard/backups';
  const scheduleStore = createScheduleStore(getDefaultScheduleIndexPath(backupDir));

  // Stop existing job
  stopCronJob(id);

  // Update schedule
  await scheduleStore.update(id, { enabled: false });

  const schedule = await scheduleStore.get(id);
  if (!schedule) {
    throw new Error(`Schedule not found: ${id}`);
  }

  return schedule;
}

export async function listSchedules(
  options?: ScheduleOptions,
): Promise<ScheduleTask[]> {
  const backupDir = options?.backupDir || '.clawguard/backups';
  const scheduleStore = createScheduleStore(getDefaultScheduleIndexPath(backupDir));

  return await scheduleStore.list();
}

export async function setRetainDays(
  id: string,
  days: number,
  options?: ScheduleOptions,
): Promise<ScheduleTask> {
  const backupDir = options?.backupDir || '.clawguard/backups';
  const scheduleStore = createScheduleStore(getDefaultScheduleIndexPath(backupDir));

  await scheduleStore.update(id, { retainDays: days });

  const schedule = await scheduleStore.get(id);
  if (!schedule) {
    throw new Error(`Schedule not found: ${id}`);
  }

  return schedule;
}

export async function deleteSchedule(
  id: string,
  options?: ScheduleOptions,
): Promise<void> {
  const backupDir = options?.backupDir || '.clawguard/backups';
  const scheduleStore = createScheduleStore(getDefaultScheduleIndexPath(backupDir));

  // Stop existing job
  stopCronJob(id);

  // Delete schedule
  await scheduleStore.delete(id);
}

export async function deleteOldBackups(
  retainDays: number,
  options?: ScheduleOptions,
): Promise<number> {
  const backupDir = options?.backupDir || '.clawguard/backups';
  const metadataStore = createMetadataStore(getDefaultMetadataIndexPath(backupDir));

  const cutoffTime = Date.now() - (retainDays * 24 * 60 * 60 * 1000);
  const backups = await metadataStore.list();

  let deletedCount = 0;
  for (const backup of backups) {
    if (backup.timestamp < cutoffTime) {
      await metadataStore.delete(backup.id);
      deletedCount++;
    }
  }

  return deletedCount;
}

export function initializeSchedules(backupDir?: string): void {
  const dir = backupDir || '.clawguard/backups';
  const scheduleStore = createScheduleStore(getDefaultScheduleIndexPath(dir));

  // This would normally be called on app startup
  // For now, we'll just load schedules without starting jobs
  scheduleStore.list().then(schedules => {
    schedules.forEach(schedule => {
      if (schedule.enabled) {
        startCronJob(schedule.id, schedule, dir);
      }
    });
  }).catch(() => {
    // Ignore errors during initialization
  });
}

function startCronJob(id: string, schedule: ScheduleTask, backupDir: string): void {
  // Stop existing job if any
  stopCronJob(id);

  // Create new cron job
  const job = new CronJob(schedule.cron, async () => {
    try {
      // Run backup
      await backup({
        level: schedule.level,
        name: `${schedule.name} - Auto`,
        output: backupDir,
      });

      // Update schedule store
      const scheduleStore = createScheduleStore(getDefaultScheduleIndexPath(backupDir));
      await scheduleStore.update(id, {
        lastRun: Date.now(),
      });

      // Delete old backups
      await deleteOldBackups(schedule.retainDays, { backupDir });
    } catch {
      // Log error but continue
    }
  }, {
    scheduled: true,
  });

  job.start();
  activeJobs.set(id, job);
}

function stopCronJob(id: string): void {
  const job = activeJobs.get(id);
  if (job) {
    job.stop();
    activeJobs.delete(id);
  }
}

function generateScheduleId(): string {
  return `sched-${Date.now()}-${crypto.randomBytes(4).toString('hex').slice(0, 8)}`;
}

export function shutdownAllSchedules(): void {
  activeJobs.forEach(job => job.stop());
  activeJobs.clear();
}
```

// packages/cli/src/commands/schedule.ts
import {
  enableSchedule,
  disableSchedule,
  listSchedules,
  setRetainDays,
  deleteSchedule,
  initializeSchedules,
  shutdownAllSchedules
} from '../../core/src/schedule/index.js';

export interface ScheduleCommandOptions {
  backupDir?: string;
}

export async function handleScheduleEnableCommand(
  name: string,
  cron: string,
  retainDays: string,
  options: ScheduleCommandOptions,
): Promise<void> {
  const retainDaysNum = parseInt(retainDays, 10);
  if (isNaN(retainDaysNum) || retainDaysNum < 1) {
    console.error('Invalid retain days. Must be a positive number.');
    process.exit(1);
  }

  const schedule = await enableSchedule({
    name,
    level: 'config', // Default to config level
    cron,
    retainDays: retainDaysNum,
    backupDir: options.backupDir,
  });

  console.log(`Schedule enabled: ${schedule.id}`);
  console.log(`Name: ${schedule.name}`);
  console.log(`Cron: ${schedule.cron}`);
  console.log(`Retain: ${schedule.retainDays} days`);
}

export async function handleScheduleDisableCommand(
  id: string,
  options: ScheduleCommandOptions,
): Promise<void> {
  const schedule = await disableSchedule(id, options);
  console.log(`Schedule disabled: ${schedule.id}`);
}

export async function handleScheduleRetainCommand(
  id: string,
  days: string,
  options: ScheduleCommandOptions,
): Promise<void> {
  const daysNum = parseInt(days, 10);
  if (isNaN(daysNum) || daysNum < 1) {
    console.error('Invalid retain days. Must be a positive number.');
    process.exit(1);
  }

  const schedule = await setRetainDays(id, daysNum, options);
  console.log(`Schedule ${schedule.id} retain days set to: ${daysNum}`);
}

export async function handleScheduleListCommand(
  options: ScheduleCommandOptions,
): Promise<void> {
  const schedules = await listSchedules(options);

  if (schedules.length === 0) {
    console.log('No schedules found.');
    return;
  }

  console.log(`Found ${schedules.length} schedule(s):\n`);
  schedules.forEach(schedule => {
    const status = schedule.enabled ? 'Enabled' : 'Disabled';
    const lastRun = schedule.lastRun
      ? new Date(schedule.lastRun).toISOString()
      : 'Never';
    const nextRun = new Date(schedule.nextRun).toISOString();

    console.log(`ID: ${schedule.id}`);
    console.log(`  Name: ${schedule.name}`);
    console.log(`  Status: ${status}`);
    console.log(`  Level: ${schedule.level}`);
    console.log(`  Cron: ${schedule.cron}`);
    console.log(`  Retain: ${schedule.retainDays} days`);
    console.log(`  Last Run: ${lastRun}`);
    console.log(`  Next Run: ${nextRun}`);
    console.log('');
  });
}

export async function handleScheduleDeleteCommand(
  id: string,
  options: ScheduleCommandOptions,
): Promise<void> {
  await deleteSchedule(id, options);
  console.log(`Schedule deleted: ${id}`);
}

export function initializeDaemon(options: ScheduleCommandOptions): void {
  initializeSchedules(options.backupDir);

  // Handle graceful shutdown
  process.on('SIG', shutdownAllSchedules);
  process.on('exit', shutdownAllSchedules);
}
